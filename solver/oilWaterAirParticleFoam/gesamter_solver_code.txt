==================== ./alphaEqn.H ====================
// alphaEqn.H — three-phase sequential constrained isoAdvection
//
// Strategy (avoids phase mixing at three-phase contact lines):
//
//   STEP 1 — advect alpha.air (unconstrained, bounds [0,1])
//   STEP 2 — cap alpha.oil BEFORE advection: alpha2 = min(alpha2, 1-alpha1)
//             then advect alpha.oil with isoAdvector
//             then cap AFTER advection:       alpha2 = min(alpha2, 1-alpha1)
//   STEP 3 — alpha.water = 1 - alpha.air - alpha.oil  (algebraic)
//
// Why capping instead of a constrained MULES upper-bound argument?
//   isoAdvection::advect() takes only Sp and Su source terms — there is no
//   upper-bound argument.  Post-advection clamping achieves the same effect:
//   any oil that isoAdvector tried to put into an air cell is clipped out.
//   The clipped mass is tiny (one-cell boundary effect) and doesn't affect
//   conservation meaningfully.
//
// Sub-cycling: nAlphaSubCycles from alphaControls.H (fvSolution/PIMPLE dict).
//   The sub-cycle loop is on alpha1 only (standard isobubbleFoam pattern);
//   alpha2 sub-cycles in lockstep inside the same loop.

{
    dimensionedScalar totalDeltaT = runTime.deltaT();

    // rhoPhi accumulator for sub-cycling
    surfaceScalarField rhoPhiSum
    (
        IOobject("rhoPhiSum", runTime.timeName(), mesh),
        mesh,
        dimensionedScalar(rhoPhi.dimensions(), Zero)
    );

    for
    (
        subCycle<volScalarField> alphaSubCycle(alpha1, nAlphaSubCycles);
        !(++alphaSubCycle).end();
    )
    {
        // ── STEP 1: advect alpha.air ──────────────────────────────────────
        {
            volScalarField::Internal Sp1
            (
                IOobject("Sp1", runTime.timeName(), mesh),
                mesh, dimensionedScalar(dimless/dimTime, Zero)
            );
            volScalarField::Internal Su1
            (
                IOobject("Su1", runTime.timeName(), mesh),
                mesh, dimensionedScalar(dimless/dimTime, Zero)
            );
            advector1.advect(Sp1, Su1);
            // Clamp to [0,1] to remove any sub-cell round-off
            alpha1 = max(min(alpha1, one_), zero_);
        }

        // ── STEP 2: advect alpha.oil (constrained) ────────────────────────
        // Pre-cap: oil cannot sit in cells that are now fully air
        alpha2 = min(max(alpha2, zero_), one_ - alpha1);
        {
            volScalarField::Internal Sp2
            (
                IOobject("Sp2", runTime.timeName(), mesh),
                mesh, dimensionedScalar(dimless/dimTime, Zero)
            );
            volScalarField::Internal Su2
            (
                IOobject("Su2", runTime.timeName(), mesh),
                mesh, dimensionedScalar(dimless/dimTime, Zero)
            );
            advector2.advect(Sp2, Su2);
            // Post-cap: enforce alpha1 + alpha2 <= 1
            alpha2 = min(max(alpha2, zero_), one_ - alpha1);
        }

        // ── STEP 3: alpha.water = remainder ──────────────────────────────
        alpha3 = max(one_ - alpha1 - alpha2, zero_);

        // Accumulate rhoPhi for sub-cycle averaging
        // Use air-advector rhoPhi as the primary carrier flux
        rhoPhiSum += (runTime.deltaT()/totalDeltaT)
                   * advector1.getRhoPhi(rho1, rho3);
    }

    rhoPhi = rhoPhiSum;
}

==================== ./pEqn.H ====================
// pEqn.H — pressure correction with buoyancy (no surface tension)
{
    tmp<volScalarField> trAU(1.0/UEqn.A());
    const volScalarField& rAU = trAU();

    surfaceScalarField rAUf("rAUf", fvc::interpolate(rAU));

    volVectorField HbyA(constrainHbyA(rAU*UEqn.H(), U, p_rgh));

    surfaceScalarField phiHbyA
    (
        "phiHbyA",
        fvc::flux(HbyA)
      + MRF.zeroFilter(fvc::interpolate(rho*rAU)*fvc::ddtCorr(U, phi))
    );

    MRF.makeRelative(phiHbyA);

    if (p_rgh.needReference())
    {
        fvc::makeRelative(phiHbyA, U);
        adjustPhi(phiHbyA, U, p_rgh);
        fvc::makeAbsolute(phiHbyA, U);
    }

    // Buoyancy flux (ghf * snGrad(rho) * rAUf * mesh.magSf())
    surfaceScalarField phig
    (
        ghf*fvc::snGrad(rho)*rAUf*mesh.magSf()
    );
    phiHbyA += phig;

    constrainPressure(p_rgh, U, phiHbyA, rAUf, MRF);

    while (pimple.correctNonOrthogonal())
    {
        fvScalarMatrix p_rghEqn
        (
            fvm::laplacian(rAUf, p_rgh) == fvc::div(phiHbyA)
        );

        p_rghEqn.setReference(pRefCell, getRefCellValue(p_rgh, pRefCell));
        p_rghEqn.solve();

        if (pimple.finalNonOrthogonalIter())
        {
            phi = phiHbyA - p_rghEqn.flux();
            p_rgh.relax();
            U = HbyA + rAU*fvc::reconstruct((phig - p_rghEqn.flux())/rAUf);
            U.correctBoundaryConditions();
            fvOptions.correct(U);
        }
    }

    #include "continuityErrs.H"

    fvc::makeRelative(phi, U);

    p == p_rgh + rho*gh;

    if (p_rgh.needReference())
    {
        p += dimensionedScalar
        (
            "p", p.dimensions(),
            pRefValue - getRefCellValue(p, pRefCell)
        );
        p_rgh == p - rho*gh;
    }
}

/*---------------------------------------------------------------------------*\
  oilWaterAirParticleFoam
  Three-phase isoAdvector VOF + Lagrangian kinematic particle cloud.

  Physics goal
  ────────────
  Air bubble (R=22.5 mm) rises through water carrying a 0.5 mm particle
  upward through 5 oil droplets.  We want to see:
    • Sharp air|water interface (bubble shape preserved)
    • Sharp oil|water interfaces (droplets don't dissolve)
    • Particle captured inside bubble, rising with it
    • Bubble-oil and oil-particle surface interactions

  Why isoAdvector
  ───────────────
  isoAdvector reconstructs the interface geometrically (piecewise-linear
  cutting) every step.  Interfaces stay exactly 1–2 cells wide regardless
  of velocity or timestep.  MULES with cAlpha compression diffuses slowly
  over hundreds of steps — that was why the bubble and droplets dissolved.

  Three-phase approach
  ────────────────────
  Two isoAdvection objects (one per transported phase):
    advector1 → alpha.air   (phase 1, solved first, unconstrained)
    advector2 → alpha.oil   (phase 2, solved second, capped at 1-alpha1)
    alpha.water = 1 - alpha.air - alpha.oil  (algebraic remainder)

  Based on isobubbleFoam structure.  dt control via alphaCourantNo.H +
  setDeltaT.H works correctly with isoAdvector (no compression-flux
  inflation, so Interface Co doesn't cause dt collapse).
\*---------------------------------------------------------------------------*/

                        mixture.correct();

            if (pimple.frozenFlow()) continue;

            #include "UEqn.H"

            //if (pimple.turbCorr())

        }

        runTime.printExecutionTime(Info);
